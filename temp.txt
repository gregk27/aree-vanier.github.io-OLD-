#include "FastLED.h"
#define NUM_LEDS 100
#define DATA_PIN 6

// Define the array of leds
CRGB leds[NUM_LEDS];

//Delay Between Images
int frameDelay = 100;
// Function to reverse every second row of LEDs, as in the sign board:
int serpentine(int i) {
    int r=i/10;
    int c=i%10;
    if ((r%2)==1)
        c=9-c;
    return 10*r+c;
}
void setup() {
    FastLED.addLeds<WS2811, DATA_PIN, RGB>(leds, NUM_LEDS);
